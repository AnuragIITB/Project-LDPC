\section{ Notations }
\begin{frame}[t]
\frametitle{ Notations }
\begin{itemize}
\item n - Block size.
\item Eb/No - Input SNR in db to maintain input BER between $10^{-2}$ to $10^{-3}$.
\item BER(IN) - Input bit error rate.
\item BER(OUT) - Output bit error rate.
\item CDB - Number of correctly decoded blocks.  
\item Itr - Average number of iterations per block. (out of 25)
\end{itemize}
\end{frame}
